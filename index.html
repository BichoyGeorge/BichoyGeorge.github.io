-- Recursive Common Table Expression (CTE) example for Employee Hierarchy Mapping
WITH OrgHierarchy AS (
    SELECT EmployeeID, ManagerID, FirstName, Title, 1 AS OrgLevel
    FROM EnterpriseEmployees
    WHERE ManagerID IS NULL
    UNION ALL
    SELECT e.EmployeeID, e.ManagerID, e.FirstName, e.Title, o.OrgLevel + 1
    FROM EnterpriseEmployees e
    INNER JOIN OrgHierarchy o ON e.ManagerID = o.EmployeeID
)
SELECT * FROM OrgHierarchy ORDER BY OrgLevel;
-- Analytical Window Function for Rolling Risk Margin Calculation
SELECT 
    TransactionDate,
    DepartmentID,
    DailyRevenue,
    SUM(DailyRevenue) OVER(PARTITION BY DepartmentID ORDER BY TransactionDate ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) AS Rolling7DayRevenue,
    DENSE_RANK() OVER(PARTITION BY DepartmentID ORDER BY DailyRevenue DESC) AS RevenueRank
FROM CorporateFactSales;
-- Optimization Pattern replacing heavy subqueries with indexed structural JOIN operations
SELECT Header.SalesOrderID, Header.OrderDate, Customer.CompanyCode
FROM SalesOrderHeader Header
INNER JOIN CorporateCustomers Customer 
    ON Header.CustomerID = Customer.CustomerID
WHERE EXISTS (
    SELECT 1 FROM SalesOrderDetail Detail 
    WHERE Detail.SalesOrderID = Header.SalesOrderID AND Detail.UnitPrice > 250
);
